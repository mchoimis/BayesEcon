1% log pdf of normal
 2% x = normal variates
 3% mu = vector of means
 4% sig2vec = vector of variances
 5function [retf] = lnpdfn(x, mu, sig2vec)
 6
 7sig2vec = real(sig2vec);
 8if minc(sig2vec) < 0
 9    error('sigma should be positive');
10end
11c = -0.5*log(2*sig2vec*pi);
12e = x - mu;
13e2 = e.*e;
14
15retf = c - 0.5*e2./sig2vec;
16end
